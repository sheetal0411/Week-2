
Reverse an Array
Write a function that reverses the elements of an array.
Example:


function reverseArray(arr) {
  let reversed = [];
  for (let i = arr.length - 1; i >= 0; i--) { 
     reversed.push(arr[i]);
  }
  return reversed;
}
console.log(reverseArray([1, 2, 3, 4]));  // Output: [4, 3, 2, 1]
